import { useState, useEffect, useRef, useCallback } from 'react'
import axios from 'axios'
import './App.css'

// Импортируем SVG иконки
import downloadIcon from './assets/download-svg.svg'

function App() {
  const [tracks, setTracks] = useState([])
  const [currentTrack, setCurrentTrack] = useState(null)
  const [isPlaying, setIsPlaying] = useState(false)
  const [progress, setProgress] = useState(0)
  const [volume, setVolume] = useState(1)
  const [currentTime, setCurrentTime] = useState(0)
  const [duration, setDuration] = useState(0)
  const [isMuted, setIsMuted] = useState(false)
  const [prevVolume, setPrevVolume] = useState(1)
  const [isLoading, setIsLoading] = useState(true)
  const [error, setError] = useState(null)
  const [isDarkTheme, setIsDarkTheme] = useState(false)
  const [loopPlaylist, setLoopPlaylist] = useState(true) // Цикл проигрывания всего списка
  const [loopTrack, setLoopTrack] = useState(false) // Цикл проигрывания одного трека
  
  const audioRef = useRef(null)
  
  // Базовый URL для бэкенда
  const API_URL = import.meta.env.VITE_API_URL || 'http://localhost:5000/api'
  
  // Текущий год для футера
  const currentYear = new Date().getFullYear();
  
  // Загружаем список треков при старте
  useEffect(() => {
    let isMounted = true
    
    const fetchTracks = async () => {
      try {
        setIsLoading(true)
        setError(null)
        const response = await axios.get(`${API_URL}/tracks`)
        if (isMounted) {
          // Добавляем лайки к трекам
          const tracksWithLikes = response.data.map(track => ({
            ...track,
            liked: false
          }))
          setTracks(tracksWithLikes)
          if (tracksWithLikes.length > 0) {
            setCurrentTrack(prev => prev || tracksWithLikes[0])
          }
          setIsLoading(false)
        }
      } catch (error) {
        console.error('Error loading tracks:', error)
        if (isMounted) {
          setError('Не удалось загрузить треки. Проверьте бэкенд.')
          setIsLoading(false)
        }
      }
    }
    
    fetchTracks()
    
    return () => {
      isMounted = false
    }
  }, [API_URL])

  // Тема оформления
  useEffect(() => {
    if (isDarkTheme) {
      document.body.classList.add('dark-theme')
    } else {
      document.body.classList.remove('dark-theme')
    }
  }, [isDarkTheme])

  // Функция для перезагрузки треков
  const reloadTracks = async () => {
    try {
      setIsLoading(true)
      const response = await axios.get(`${API_URL}/tracks`)
      const tracksWithLikes = response.data.map(track => ({
        ...track,
        liked: track.liked || false,
        cover: track.cover || `https://picsum.photos/seed/${track.id}/300/300`
      }))
      setTracks(tracksWithLikes)
      if (tracksWithLikes.length > 0 && !currentTrack) {
        setCurrentTrack(tracksWithLikes[0])
      }
      setIsLoading(false)
      setError(null)
    } catch (error) {
      console.error('Error reloading tracks:', error)
      setError('Ошибка при обновлении списка треков')
      setIsLoading(false)
    }
  }

  const handleTrackSelect = useCallback((track) => {
    setCurrentTrack(track)
    setIsPlaying(true)
    
    if (audioRef.current) {
      audioRef.current.src = `${API_URL}/play/${track.file}`
      audioRef.current.volume = isMuted ? 0 : volume
      audioRef.current.load()
      audioRef.current.play().catch(e => console.log("Autoplay blocked:", e))
    }
  }, [API_URL, isMuted, volume])
  
  const togglePlayPause = () => {
    if (!currentTrack) return
    
    if (!audioRef.current.src && currentTrack.file) {
      audioRef.current.src = `${API_URL}/play/${currentTrack.file}`
      audioRef.current.volume = isMuted ? 0 : volume
    }
    
    if (isPlaying) {
      audioRef.current.pause()
    } else {
      audioRef.current.play().catch(e => {
        console.log("Play failed:", e)
        setIsPlaying(false)
      })
    }
    setIsPlaying(!isPlaying)
  }
  
  const handleTimeUpdate = () => {
    if (audioRef.current) {
      const currentTimeValue = audioRef.current.currentTime
      const durationValue = audioRef.current.duration || 0
      const progressPercent = durationValue > 0 ? (currentTimeValue / durationValue) * 100 : 0
      setCurrentTime(currentTimeValue)
      setDuration(durationValue)
      setProgress(progressPercent)
    }
  }
  
  const handleLoadedMetadata = () => {
    if (audioRef.current) {
      setDuration(audioRef.current.duration || 0)
      setProgress(0)
      setCurrentTime(0)
    }
  }
  
  const handleProgressClick = (e) => {
    if (!audioRef.current || !currentTrack || duration === 0) return
    
    const progressBar = e.currentTarget
    const clickPosition = e.clientX - progressBar.getBoundingClientRect().left
    const progressBarWidth = progressBar.clientWidth
    const percent = clickPosition / progressBarWidth
    
    const newTime = duration * percent
    audioRef.current.currentTime = newTime
    setCurrentTime(newTime)
    setProgress(percent * 100)
  }
  
  const toggleMute = () => {
    if (isMuted) {
      // Включаем звук, восстанавливаем предыдущую громкость
      setIsMuted(false)
      setVolume(prevVolume)
      if (audioRef.current) {
        audioRef.current.volume = prevVolume
      }
    } else {
      // Выключаем звук, запоминаем текущую громкость
      setIsMuted(true)
      setPrevVolume(volume)
      setVolume(0)
      if (audioRef.current) {
        audioRef.current.volume = 0
      }
    }
  }
  
  const handleVolumeChange = (e) => {
    const newVolume = parseFloat(e.target.value)
    setVolume(newVolume)
    
    if (newVolume === 0) {
      setIsMuted(true)
    } else if (isMuted) {
      setIsMuted(false)
      setPrevVolume(newVolume)
    }
    
    if (audioRef.current) {
      audioRef.current.volume = newVolume
    }
  }
  
  const handleNextTrack = useCallback(() => {
    if (!currentTrack || tracks.length === 0) return
    
    const currentIndex = tracks.findIndex(t => t.id === currentTrack.id)
    const nextIndex = (currentIndex + 1) % tracks.length
    handleTrackSelect(tracks[nextIndex])
  }, [currentTrack, tracks, handleTrackSelect])
  
  const handlePrevTrack = useCallback(() => {
    if (!currentTrack || tracks.length === 0) return
    
    const currentIndex = tracks.findIndex(t => t.id === currentTrack.id)
    const prevIndex = currentIndex === 0 ? tracks.length - 1 : currentIndex - 1
    handleTrackSelect(tracks[prevIndex])
  }, [currentTrack, tracks, handleTrackSelect])

  // Скачивание трека - исправленная версия
  const handleDownloadTrack = async (track, e) => {
    e.stopPropagation()
    if (track && track.file) {
      try {
        // Создаем URL для скачивания
        const downloadUrl = `${API_URL}/play/${track.file}`
        
        // Используем fetch для получения файла как blob
        const response = await fetch(downloadUrl)
        if (!response.ok) {
          throw new Error(`Ошибка загрузки: ${response.status}`)
        }
        
        // Получаем данные как blob
        const blob = await response.blob()
        
        // Создаем URL для скачивания из blob
        const blobUrl = window.URL.createObjectURL(blob)
        
        // Создаем временную ссылку для скачивания
        const link = document.createElement('a')
        link.href = blobUrl
        link.download = track.file // Имя файла для сохранения
        link.style.display = 'none'
        
        // Добавляем в документ и эмулируем клик
        document.body.appendChild(link)
        link.click()
        
        // Убираем ссылку из документа
        document.body.removeChild(link)
        
        // Освобождаем URL объекта blob
        window.URL.revokeObjectURL(blobUrl)
        
        // Уведомляем пользователя
        console.log(`Трек "${track.title}" скачивается`)
      } catch (error) {
        console.error('Ошибка при скачивании трека:', error)
        alert('Не удалось скачать трек. Проверьте подключение к серверу.')
      }
    }
  }

  // Форматируем время в MM:SS
  const formatTime = (seconds) => {
    if (isNaN(seconds) || seconds === 0) return '0:00'
    const mins = Math.floor(seconds / 60)
    const secs = Math.floor(seconds % 60)
    return `${mins}:${secs.toString().padStart(2, '0')}`
  }

  // Обработчики кнопок header
  const handleFeature1 = () => {
    alert('Функция 1: В разработке...')
  }

  const handleFeature2 = () => {
    alert('Функция 2: Скоро будет доступно!')
  }

  const toggleTheme = () => {
    setIsDarkTheme(!isDarkTheme)
  }

  const handleAccount = () => {
    alert('Вход в аккаунт: Система авторизации в разработке')
  }

  // Функция для лайка трека
  const handleLikeTrack = (trackId, e) => {
    e.stopPropagation() // Останавливаем всплытие, чтобы не выбирался трек
    
    setTracks(prevTracks => 
      prevTracks.map(track => 
        track.id === trackId ? { ...track, liked: !track.liked } : track
      )
    )
    
    if (currentTrack && currentTrack.id === trackId) {
      setCurrentTrack(prev => ({ ...prev, liked: !prev.liked }))
    }
  }

  // Переключение режима повтора плейлиста
  const toggleLoopPlaylist = () => {
    setLoopPlaylist(!loopPlaylist)
    if (!loopPlaylist && loopTrack) {
      // Если включаем повтор плейлиста, выключаем повтор трека
      setLoopTrack(false)
    }
  }

  // Переключение режима повтора одного трека
  const toggleLoopTrack = () => {
    setLoopTrack(!loopTrack)
    if (!loopTrack && loopPlaylist) {
      // Если включаем повтор трека, выключаем повтор плейлиста
      setLoopPlaylist(false)
    }
  }

  // Обработка окончания трека с учетом режимов повтора
  useEffect(() => {
    const audioElement = audioRef.current
    if (!audioElement) return

    const handleTrackEnd = () => {
      if (loopTrack && currentTrack) {
        // Если включен повтор трека, просто перезапускаем текущий
        audioElement.currentTime = 0
        audioElement.play().catch(e => console.log("Play failed:", e))
      } else if (loopPlaylist) {
        // Если включен повтор плейлиста, переходим к следующему
        handleNextTrack()
      } else {
        // Если оба режима выключены, просто останавливаем
        setIsPlaying(false)
      }
    }

    audioElement.addEventListener('ended', handleTrackEnd)

    return () => {
      audioElement.removeEventListener('ended', handleTrackEnd)
    }
  }, [currentTrack, loopPlaylist, loopTrack, handleNextTrack])

  return (
    <div className="App">
      {/* НОВЫЙ HEADER С 3-мя частями */}
      <header className="App-header">
        {/* Левая часть - логотип */}
        <div className="header-left">
          <div className="logo">DrD Life</div>
        </div>
        
        {/* Центральная часть - название плеера */}
        <div className="header-center">
          <h1>🎵 Мой Музыкальный Плеер (Vite + React)</h1>
          <p className="header-subtitle">Лёгкий и быстрый плеер</p>
        </div>
        
        {/* Правая часть - кнопки */}
        <div className="header-right">
          <button 
            onClick={handleFeature1} 
            className="header-btn"
            title="Функция 1"
          >
            <span className="btn-icon">⚡</span>
            <span className="btn-text">Функция 1</span>
          </button>
          
          <button 
            onClick={handleFeature2} 
            className="header-btn"
            title="Функция 2"
          >
            <span className="btn-icon">🔧</span>
            <span className="btn-text">Функция 2</span>
          </button>
          
          <button 
            onClick={toggleTheme} 
            className="header-btn theme-btn"
            title={isDarkTheme ? 'Переключить на светлую тему' : 'Переключить на темную тему'}
          >
            <span className="btn-icon">{isDarkTheme ? '☀️' : '🌙'}</span>
            <span className="btn-text">{isDarkTheme ? 'Светлая' : 'Тёмная'}</span>
          </button>
          
          <button 
            onClick={handleAccount} 
            className="header-btn account-btn small"
            title="Аккаунт"
          >
            <span className="btn-icon">👤</span>
          </button>
        </div>
      </header>
      
      <main className="player-container">
        {/* Основной плеер */}
        <div className="main-player">
          {currentTrack ? (
            <>
              <div className="player-content">
                {/* Левый блок - обложка в фиксированном контейнере */}
                <div className="album-art-container">
                  <div className="album-art">
                    {currentTrack.cover ? (
                      <div className={`album-art-img ${!isPlaying ? 'small' : ''}`}>
                        <img 
                          src={`${API_URL}/cover/${currentTrack.cover}`} 
                          alt="Обложка альбома" 
                          onError={(e) => {
                            e.target.onerror = null
                            e.target.parentElement.innerHTML = `
                              <div class="album-art-placeholder ${!isPlaying ? 'small' : ''}">
                                🎵
                                <div class="album-hint">Обложка альбома</div>
                              </div>
                            `
                          }}
                        />
                      </div>
                    ) : (
                      <div className={`album-art-placeholder ${!isPlaying ? 'small' : ''}`}>
                        🎵
                        <div className="album-hint">Обложка альбома</div>
                      </div>
                    )}
                  </div>
                </div>
                
                {/* Правый блок - управление */}
                <div className="player-controls">
                  {/* Верхняя часть: информация о треке */}
                  <div className="track-header">
                    <div className="track-title-section">
                      <h2 className="track-title">{currentTrack.title}</h2>
                      <p className="track-artist">{currentTrack.artist}</p>
                    </div>
                    <div className="track-header-right">
                      <button 
                        className={`like-btn ${currentTrack.liked ? 'liked' : ''}`}
                        onClick={(e) => handleLikeTrack(currentTrack.id, e)}
                        title={currentTrack.liked ? 'Убрать из избранного' : 'Добавить в избранное'}
                      >
                        {currentTrack.liked ? '❤️' : '🤍'}
                      </button>
                      <div className="track-id-section">
                        <span className="track-id-label">ID</span>
                        <span className="track-id-value">{currentTrack.id}</span>
                      </div>
                    </div>
                  </div>
                  
                  {/* Средняя часть: прогресс бар */}
                  <div className="progress-section">
                    <div 
                      className="progress-bar" 
                      onClick={handleProgressClick}
                    >
                      <div 
                        className="progress-fill" 
                        style={{ width: `${progress}%` }}
                      ></div>
                    </div>
                    <div className="time-display">
                      <span className="current-time">{formatTime(currentTime)}</span>
                      <span className="duration">{formatTime(duration)}</span>
                    </div>
                  </div>
                  
                  {/* Нижняя часть: кнопки управления и громкость */}
                  <div className="player-actions">
                    {/* Кнопки управления треком */}
                    <div className="playback-controls">
                      <button onClick={handlePrevTrack} className="control-btn" title="Предыдущий трек">
                        ⏮️
                      </button>
                      
                      <button onClick={togglePlayPause} className="play-btn" title={isPlaying ? 'Остановить' : 'Воспроизвести'}>
                        {isPlaying ? '⏸️ Остановить' : '▶️ Воспроизвести'}
                      </button>
                      
                      <button onClick={handleNextTrack} className="control-btn" title="Следующий трек">
                        ⏭️
                      </button>
                    </div>
                    
                    {/* Кнопка повтора текущего трека */}
                    <button 
                      onClick={toggleLoopTrack} 
                      className={`control-btn ${loopTrack ? 'active' : ''}`}
                      title={loopTrack ? 'Выключить повтор трека' : 'Включить повтор трека'}
                    >
                      {loopTrack ? '🔂' : '🔁'}
                    </button>
                    
                    {/* Регулировка громкости */}
                    <div className="volume-control">
                      <button 
                        onClick={toggleMute} 
                        className="volume-toggle-btn"
                        title={isMuted ? 'Включить звук' : 'Выключить звук'}
                      >
                        {isMuted ? '🔇' : volume > 0.5 ? '🔊' : '🔉'}
                      </button>
                      
                      <input
                        type="range"
                        min="0"
                        max="1"
                        step="0.01"
                        value={isMuted ? 0 : volume}
                        onChange={handleVolumeChange}
                        className="volume-slider"
                        title="Громкость"
                      />
                      
                      <span className="volume-percent">
                        {isMuted ? '0%' : `${Math.round(volume * 100)}%`}
                      </span>
                    </div>
                  </div>
                </div>
              </div>
              
              {/* Скрытый audio элемент */}
              <audio
                ref={audioRef}
                onTimeUpdate={handleTimeUpdate}
                onLoadedMetadata={handleLoadedMetadata}
                onError={(e) => console.error("Audio error:", e)}
              />
            </>
          ) : (
            <div className="no-track">
              {isLoading ? (
                <>
                  <p>⏳ Загрузка треков...</p>
                  <div className="loading-spinner"></div>
                </>
              ) : error ? (
                <>
                  <p>❌ {error}</p>
                  <button onClick={reloadTracks} className="refresh-btn">
                    Попробовать снова
                  </button>
                </>
              ) : (
                <>
                  <p>🎵 Выберите трек для воспроизведения</p>
                  <button onClick={reloadTracks} className="refresh-btn">
                    Обновить список треков
                  </button>
                </>
              )}
            </div>
          )}
        </div>
        
        {/* Список треков */}
        <div className="track-list">
          <div className="track-list-header">
            <h3>Библиотека треков ({tracks.length})</h3>
            <div className="track-list-actions">
              <button 
                onClick={toggleLoopPlaylist} 
                className={`loop-playlist-btn ${loopPlaylist ? 'active' : ''}`}
                title={loopPlaylist ? 'Проигрывать только выбранный трек' : 'Зациклить весь список треков'}
              >
                {loopPlaylist ? '🔁 Весь список' : '▶️ Только выбранный'}
              </button>
              <button onClick={reloadTracks} className="refresh-btn small" title="Обновить список треков">
                🔄
              </button>
            </div>
          </div>
          
          {isLoading ? (
            <div className="loading-tracks">
              <p>⏳ Загрузка списка треков...</p>
            </div>
          ) : error ? (
            <div className="error-message">
              <p>❌ {error}</p>
              <button onClick={reloadTracks} className="refresh-btn small">
                Повторить
              </button>
            </div>
          ) : tracks.length === 0 ? (
            <div className="empty-list">
              <p>📭 Нет доступных треков</p>
              <p className="hint">Добавьте аудиофайлы в папку backend/music_files/</p>
              <p className="hint">Поддерживаемые форматы: MP3, WAV, OGG, FLAC, M4A, AAC</p>
              <button onClick={reloadTracks} className="refresh-btn small">
                Проверить снова
              </button>
            </div>
          ) : (
            <ul className="tracks">
              {tracks.map(track => (
                <li 
                  key={track.id} 
                  className={`track-item ${currentTrack?.id === track.id ? 'active' : ''}`}
                  onClick={() => handleTrackSelect(track)}
                >
                  <span className="track-number">{track.id}</span>
                  <div className="track-cover">
                    {track.cover ? (
                      <img 
                        src={`${API_URL}/cover/${track.cover}`} 
                        alt={track.title}
                        onError={(e) => {
                          e.target.onerror = null
                          e.target.parentElement.innerHTML = `
                            <div class="track-cover-placeholder">
                              🎵
                            </div>
                          `
                        }}
                      />
                    ) : (
                      <div className="track-cover-placeholder">
                        🎵
                      </div>
                    )}
                  </div>
                  <div className="track-details">
                    <strong className="track-title">{track.title}</strong>
                    <span className="track-artist">{track.artist}</span>
                  </div>
                  <div className="track-actions">
                    <div className="track-status">
                      {currentTrack?.id === track.id && isPlaying && (
                        <span className="playing-indicator">▶️ Играет</span>
                      )}
                      {currentTrack?.id === track.id && !isPlaying && (
                        <span className="paused-indicator">⏸️ На паузе</span>
                      )}
                    </div>
                    <button 
                      className={`track-like-btn ${track.liked ? 'liked' : ''}`}
                      onClick={(e) => handleLikeTrack(track.id, e)}
                      title={track.liked ? 'Убрать из избранного' : 'Добавить в избранное'}
                    >
                      {track.liked ? '❤️' : '🤍'}
                    </button>
                    <button 
                      className="track-download-btn"
                      onClick={(e) => handleDownloadTrack(track, e)}
                      title="Скачать трек"
                    >
                      <img 
                        src={downloadIcon} 
                        alt="Скачать" 
                        className="download-icon"
                      />
                    </button>
                  </div>
                </li>
              ))}
            </ul>
          )}
        </div>
      </main>
      
      {/* НОВЫЙ ФУТЕР С 3-мя частями - как хедер */}
      <footer className="footer">
        {/* Левая часть - год */}
        <div className="footer-left">
          <div className="footer-year">{currentYear}</div>
        </div>
        
        {/* Центральная часть - основной текст */}
        <div className="footer-center">
          <p>Сделано с Vite + React + Python Flask</p>
          <div className="debug-info">
            API: <code>{API_URL}</code> | 
            Треков: {tracks.length} | 
            Громкость: {isMuted ? '0%' : `${Math.round(volume * 100)}%`} |
            Тема: {isDarkTheme ? 'Тёмная' : 'Светлая'} |
            Повтор: {loopTrack ? 'Трека' : loopPlaylist ? 'Списка' : 'Нет'} |
            {isLoading && ' ⏳ Загрузка...'}
          </div>
        </div>
        
        {/* Правая часть - логотип */}
        <div className="footer-right">
          <div className="footer-logo">DrD Life</div>
        </div>
      </footer>
    </div>
  )
}

export default App
